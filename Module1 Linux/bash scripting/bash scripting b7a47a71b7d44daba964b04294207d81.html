  • while
    #!/bin/bash

sum=0
while true
do
   read -p "enter a score: " score

   if [ "$score" == "q" ]
   then
      echo " "
      echo "Quiting....."
      break
   fi

   sum=$(($sum+$score))
   echo "total score: $sum"
done

#!/bin/bash

function sum(){
   total=$(($1+$2))
   return $total
}

result=$?
# above two lines same as result=$(sum 2 10)

echo "2 + 10 = $result"

function create_file() {
   file_name=$1
   is_shell_script=$2

   touch $file_name
   echo " file $file_name created"

   if [ "$is_shell_script" = true ]
   then
      chmod u+x $file_name
      echo "added execute permission"
   fi
   echo " "
}
